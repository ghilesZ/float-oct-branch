init {
  x=[0,1];
  y=[0,1];
}

body {
  x = (3.0/4.0) * x - (1.0/8.0)* y;
  y = x;
}

  // This goal was not taken from the paper itself, but was chosen
  //   as a reasonable property to prove.
  x=[-0.2,1];

}
